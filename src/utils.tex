\usepackage{ifthen}
\newcommand\changepapersize[1]{%
    \ifthenelse{\equal{#1}{portrait}}
    {
        \defaultareasettings
    }
    {
        \cleardoublepage
        \KOMAoptions{paper=#1,DIV=last}%
        \newgeometry{margin=1in,footskip=0.5in}%
        \fancyfootoffset{0pt}
    }
}
\AtBeginDocument{\storeareas\defaultareasettings}
\BeforeRestoreareas{\clearpage}

\newcommand\landscapepage {
    \changepapersize{landscape}
}
\newcommand\portraitpage {
    \changepapersize{portrait}
}
